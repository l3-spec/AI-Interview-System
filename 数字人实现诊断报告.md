    isSpeaking: boolean,
    currentTTSStream: NodeJS.ReadableStream | null
  }>();
  
  interrupt(sessionId: string) {
    const session = this.activeSessions.get(sessionId);
    
    if (session && session.isSpeaking) {

      session.currentTTSStream?.destroy();

    this.activeSessions.set(sessionId, {
      isSpeaking: true,
      currentTTSStream: ttsStream
    });
    
    ttsStream.on('end', () => {
      const session = this.activeSessions.get(sessionId);
      if (session) {
        session.isSpeaking = false;
      }
    });
  }
  
  isSpeaking(sessionId: string): boolean {
    return this.activeSessions.get(sessionId)?.isSpeaking ?? false;
